Question 9 : Create a set of 10 random integers between 10 and 40. Now print the count of numbers less than 20 and remove numbers > 30 from the set.

import random
random_numbers={random.randint(10, 40)
for _ in range(10)}
print("Original set:",random_numbers)
num_of_elements_less_than_20=sum(1 for number in random_numbers if number<20)
print("Count of numbers less than 20:", num_of_elements_less_than_20)
num_to_remove={number for number in random_numbers if number>30}
random_numbers.difference_update(num_to_remove)
print("After removing numbers greater than 30:",random_numbers)
